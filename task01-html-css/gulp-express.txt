var express = require('express') //llamamos a Express
var OnePiceInformation = require('./OnePiceInformation.json');

var app = express()

var port = process.env.PORT || 8080  // establecemos nuestro puerto

app.get('/', function(req, res) {
    res.json({ mensaje: 'API ONE PIECE' })
})

app.get('/OnePiceInformation', function(req, res) {
    res.json({ mensaje: OnePiceInformation })
})


app.listen(port)
console.log('API escuchando en el puerto ' + port)
